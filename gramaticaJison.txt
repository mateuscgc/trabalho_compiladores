S
    : COMANDOS
    ;

COMANDOS
    : COMANDO COMANDOS
    | 
    ;

COMANDO
    : DECLARE LISTAIDS AS TIPO DOT
    | RESIZE ID TO EXPARIT DOT
    | PUT PUTAVEL IN ID INDEX DOT
    | IF EXPARIT OPRREL EXPARIT THEN OPENCOLCHETES COMANDOS CLOSECOLCHETES SENAO
    | FOR ID INDEX FROM EXPARIT TO EXPARIT DO OPENCOLCHETES COMANDOS CLOSECOLCHETES
    | FOREACH ID INDEX IN ID DO OPENCOLCHETES COMANDOS CLOSECOLCHETES
    | READ ID INDEX DOT
    | PRINT IMPRIMIVEL DOT
    ;

LISTAIDS
    : ID NINDEX COMMA LISTAIDS
    | ID NINDEX
    ;

NINDEX
    : OPENCOLCHETES CLOSECOLCHETES
    | 
    ;

INDEX
    : OPENCOLCHETES EXPARIT CLOSECOLCHETES
    | 
    ;

TIPO
    : NUMBER
    | LETTER
    ;

EXPARIT
    : EXPARIT PLUS T
    | EXPARIT MINUS T
    | T
    ;

T
    : T TIMES F
    | T DIVIDE F
    | T MOD F
    | F
    ;

F
    : OPENPARENTESES EXPARIT CLOSEPARENTESES
    | MINUS OPENPARENTESES EXPARIT CLOSEPARENTESES
    | ID INDEX
    | MINUS ID INDEX
    | CONSTNUM
    | MINUS CONSTNUM
    | CONSTCHAR
    | MINUS CONSTCHAR
    ;

PUTAVEL
    : EXPARIT
    | CONSTSTRING
    ;

SENAO
    : ELSE OPENCOLCHETES COMANDOS CLOSECOLCHETES
    | 
    ;

OPRREL
    : LESSTHAN
    | GREATERTHAN
    | LESSEQUALTHAN
    | GREATEREQUALTHAN
    | EQUAL
    | DIFERENT
    ;

IMPRIMIVEL
    : ID INDEX
    | CONSTNUM
    | CONSTCHAR
    | CONSTSTRING
    ;
